#include "DHT.h"
#define DHTTYPE DHT11

const int rosso = 10;
const int verde = 11;
const int blu = 12;
const int buzzer = 5;
const int sensore = 8;

DHT dht(sensore, DHTTYPE);

const int TempMax = 22;
const int TempMin = 14;

void setup() {
  Serial.begin(9600);
  dht.begin();
  pinMode(rosso, OUTPUT);
  pinMode(verde, OUTPUT);
  pinMode(blu, OUTPUT);
  pinMode(buzzer, OUTPUT);
}
void loop() {
 
  float h = dht.readHumidity();
  float t = dht.readTemperature();
 
  if (isnan(h) || isnan(t)) {
    return;
  }
 
  digitalWrite(rosso, LOW);
  digitalWrite(verde, LOW);
  digitalWrite(blu, LOW);
  noTone(buzzer);
 
  if (t > TempMax) {
    digitalWrite(rosso, HIGH);
    tone(buzzer, 1000);
    delay(100);
    noTone(buzzer);
    delay(100);
  }
  else if (t < TempMin) {
    digitalWrite(blu, HIGH);
    tone(buzzer, 600);
    delay(600);
    noTone(buzzer);
    delay(600);
  }
  else {
    digitalWrite(verde, HIGH);
  }
 
  Serial.print("Temperature: ");
  Serial.println(t);
  Serial.print("Humidity: ");
  Serial.println(h);
 
  delay(500);
}
